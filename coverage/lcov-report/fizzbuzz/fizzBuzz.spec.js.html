var fizzBuzz = require('./fizzBuzz');
var expect = require("expect");
 
describe('FizzBuzz', function () {
  describe("should return the number when not divisible by 3 or 5", function () {
    it('1 => "1"', function () {
      expect(fizzBuzz(1)).toBe("1");
    });
 
    it('2 => "2"', function () {
      expect(fizzBuzz(2)).toBe("2");
    });
  });
 
  describe('should return "fizz" when divisible by only 3', function () {
    it('3 => "fizz"', function () {
      expect(fizzBuzz(3)).toBe("fizz");
    });
 
    it('6 => "fizz"', function () {
      expect(fizzBuzz(6)).toBe("fizz");
    });
  });
 
  describe('should return "buzz" when divisible by only 5', function () {
    it('5 => "buzz"', function () {
      expect(fizzBuzz(5)).toBe("buzz");
    });
 
    it('10 => "buzz"', function () {
      expect(fizzBuzz(10)).toBe("buzz");
    });
  });
 
  describe('should return "fizzbuzz" when divisible by both 3 and 5', function () {
    it('15 => "fizzbuzz"', function () {
      expect(fizzBuzz(15)).toBe("fizzbuzz");
    });
 
    it('45 => "fizzbuzz"', function () {
      expect(fizzBuzz(45)).toBe("fizzbuzz");
    });
 
  });
});

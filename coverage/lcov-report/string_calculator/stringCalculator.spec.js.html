var calculate = require('./stringCalculator');
var expect = require("expect");
 
describe('String Calculator', function () {
 
  it('should return 0 given empty string', function () {
    expect(calculate('')).toBe(0);
  });
 
  it('should return 1 given "1"', function () {
    expect(calculate('1')).toBe(1);
  });
 
  it('should return 2 given "2"', function () {
    expect(calculate('2')).toBe(2);
  });
 
  it('should return 3 given "1,2"', function (){
    expect(calculate('1,2')).toBe(3);
  });
 
  it('should return 4 given "2,2"', function (){
    expect(calculate('2,2')).toBe(4);
  });
 
  it('should return 7 given "2,2,3"', function (){
    expect(calculate('2,2,3')).toBe(7);
  });
 
  describe("should support '\\n' and ',' as default delimiters", function () {
    it('should return 6 given "1\n2,3"', function (){
      expect(calculate('1\n2,3')).toBe(6);
    });
 
    it('should return 7 given "2\n2,3"', function (){
    expect(calculate('2\n2,3')).toBe(7);
    });
 
    it('should return 8 given "2\n3\n3"', function (){
      expect(calculate('2\n3\n3')).toBe(8);
    });
  });
 
  describe("should ignore numbers greater than 1000", function () {
    it('should return 2 given "2,1001"', function (){
      expect(calculate('2,1001')).toBe(2);
    });
 
    it('should return 1002 given "2,1000"', function (){
      expect(calculate('2,1000')).toBe(1002);
    });
  });
 
  describe("should support custom delimiter after //", function () {
    it('should return 3 given "//;\n1;2"', function () {
      expect(calculate('//;\n1;2')).toBe(3);
    });
 
    it('should return 5 given "//;\n3;2"', function () {
      expect(calculate('//;\n3;2')).toBe(5);
    });
 
    it('should return 5 given "//!\n3!2"', function () {
      expect(calculate('//!\n3!2')).toBe(5);
    });
  });
});

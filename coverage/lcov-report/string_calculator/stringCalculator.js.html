module.exports = function(inputString) {
 
  if(inputString === "") {
    return 0;
  }
 
  var delimiters = determineDelimiters();
  var inputs = extractInputs(delimiters);
 
  return sum(inputs);
 
  function determineDelimiters() {
    var defaultDelimiters = "\n,";
 
    if (hasCustomDelimiter()) {
      return defaultDelimiters + customDelimiter();
    }
    return defaultDelimiters;
  }
 
  function customDelimiter() {
    return inputString.substring(2, 3);
  }
 
  function hasCustomDelimiter() {
    return inputString.substring(0, 2) == "//";
  }
 
  function extractInputs(delimiters) {
    var delimiterRegEx = new RegExp("[" + delimiters + "]");
    var inputs = inputString.split(delimiterRegEx);
    return inputs;
  }
 
  function sum(inputs) {
    return inputs.reduce(addOne,0);
  }
 
  function addOne(sum, element) {
    var ourNumber = parseInt(element);
 
    if (inRange())
      return sum + ourNumber;
    else
      return sum;
 
    function inRange() {
      return ourNumber <= 1000;
    }
  }
};

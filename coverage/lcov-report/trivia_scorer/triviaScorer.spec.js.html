var scorer = require('./triviaScorer');
var expect = require('expect');
 
describe('Trivia Scorer', function () {
  it('should return 1 for "Y"',function(){
    expect(scorer('Y')).toBe(1);
  });
 
  it('should return 0 for "X"',function(){
    expect(scorer('X')).toBe(0);
  });
 
  it('should return 0 for an empty string', function () {
    expect(scorer('')).toBe(0);
  });
 
  it('should return 1 for X,Y', function () {
    expect(scorer('X,Y')).toBe(1);
  });
 
  it('should return 3 for Y,X,Y,Y', function () {
    expect(scorer('Y,X,Y,Y')).toBe(3);
  });
 
  it('should return a score of 5 for 5', function () {
    expect(scorer('5,X,Y')).toBe(6);
  });
 
  it('should return a score of 6 for 1,2,3', function () {
    expect(scorer('1,2,3')).toBe(6);
  });
 
});
